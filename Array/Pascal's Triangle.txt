class Solution {
public:
    vector<int> generatePascalRow(int row){
        vector<int> res;
        int ans = 1;
        res.push_back(ans);
        for(int i = 1; i < row; i++) {  //calculating the nCr with this loop.
            ans = ans * (row - i);
            ans = ans / i;
            res.push_back(ans);
        }
        return res;
    }
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> matrix;
        for(int i = 1; i <= numRows; i++){
            matrix.push_back(generatePascalRow(i));//generate all ith pascal rows and add.
        }
        return matrix;
    }
};
